# Quickstart

Get up and running with SochDB in under 5 minutes. This guide will walk you through installation and your first database operations.

<Steps>
  <Step title="Install SochDB">
    Choose your preferred language and install the SDK:

    <CodeGroup>
    ```bash Python
    pip install sochdb
    ```

    ```bash Node.js
    npm install @sochdb/sochdb
    ```

    ```bash Go
    go get github.com/sochdb/sochdb-go@latest
    ```

    ```bash Rust
    cargo add sochdb
    ```
    </CodeGroup>

    <Note>
    See the [installation guide](/installation) for additional options including Docker deployment.
    </Note>
  </Step>

  <Step title="Open your first database">
    Create and open a database in just one line:

    <CodeGroup>
    ```python Python
    from sochdb import Database

    # Open or create a database
    db = Database.open("./my_db")
    ```

    ```typescript Node.js
    import { Database } from '@sochdb/sochdb';

    // Open or create a database
    const db = await Database.open('./my_db');
    ```

    ```go Go
    import sochdb "github.com/sochdb/sochdb-go"

    // Open or create a database
    db, err := sochdb.Open("./my_db")
    if err != nil {
        log.Fatal(err)
    }
    defer db.Close()
    ```

    ```rust Rust
    use sochdb::Database;

    // Open or create a database
    let db = Database::open("./my_db")?;
    ```
    </CodeGroup>

    The database will be created if it doesn't exist. SochDB uses a single directory for all database files.
  </Step>

  <Step title="Store and retrieve data">
    Use the simple key-value API to store and retrieve data:

    <CodeGroup>
    ```python Python
    # Store a value
    db.put(b"greeting", b"Hello, SochDB!")

    # Retrieve the value
    value = db.get(b"greeting")
    print(value.decode())  # "Hello, SochDB!"
    ```

    ```typescript Node.js
    // Store a value
    await db.put('greeting', 'Hello, SochDB!');

    // Retrieve the value
    const value = await db.get('greeting');
    console.log(value?.toString());  // "Hello, SochDB!"
    ```

    ```go Go
    // Store a value
    err = db.Put("greeting", []byte("Hello, SochDB!"))
    if err != nil {
        log.Fatal(err)
    }

    // Retrieve the value
    value, err := db.Get("greeting")
    if err != nil {
        log.Fatal(err)
    }
    fmt.Println(string(value))  // "Hello, SochDB!"
    ```

    ```rust Rust
    // Store a value
    db.put(b"greeting", b"Hello, SochDB!")?;

    // Retrieve the value
    if let Some(value) = db.get(b"greeting")? {
        println!("{}", String::from_utf8_lossy(&value));  // "Hello, SochDB!"
    }
    ```
    </CodeGroup>
  </Step>

  <Step title="Use path-based hierarchical keys">
    SochDB supports path-based keys for organizing data hierarchically:

    <CodeGroup>
    ```python Python
    # Store hierarchical data
    db.put_path("users/alice/name", b"Alice Smith")
    db.put_path("users/alice/email", b"alice@example.com")
    db.put_path("users/bob/name", b"Bob Jones")

    # Retrieve by path
    name = db.get_path("users/alice/name")
    print(name.decode())  # "Alice Smith"

    # Scan all users
    users = list(db.scan(b"users/", b"users/~"))
    for key, value in users:
        print(f"{key.decode()}: {value.decode()}")
    ```

    ```typescript Node.js
    // Store hierarchical data
    await db.putPath(['users', 'alice', 'name'], 'Alice Smith');
    await db.putPath(['users', 'alice', 'email'], 'alice@example.com');
    await db.putPath(['users', 'bob', 'name'], 'Bob Jones');

    // Retrieve by path
    const name = await db.getPath(['users', 'alice', 'name']);
    console.log(name?.toString());  // "Alice Smith"

    // Query with prefix scan
    const results = await db.query('users/').limit(10).execute();
    for (const [key, value] of results) {
      console.log(`${key} = ${value.toString()}`);
    }
    ```

    ```go Go
    // Store hierarchical data
    db.PutPath([]string{"users", "alice", "name"}, []byte("Alice Smith"))
    db.PutPath([]string{"users", "alice", "email"}, []byte("alice@example.com"))
    db.PutPath([]string{"users", "bob", "name"}, []byte("Bob Jones"))

    // Retrieve by path
    name, _ := db.GetPath([]string{"users", "alice", "name"})
    fmt.Println(string(name))  // "Alice Smith"

    // Query prefix scan
    results, _ := db.Query("users/").Limit(10).Execute()
    for _, kv := range results {
        fmt.Printf("%s = %s\n", kv.Key, string(kv.Value))
    }
    ```

    ```rust Rust
    // Store hierarchical data
    db.put_path(&["users", "alice", "name"], b"Alice Smith")?;
    db.put_path(&["users", "alice", "email"], b"alice@example.com")?;
    db.put_path(&["users", "bob", "name"], b"Bob Jones")?;

    // Retrieve by path
    if let Some(name) = db.get_path(&["users", "alice", "name"])? {
        println!("{}", String::from_utf8_lossy(&name));  // "Alice Smith"
    }

    // Query prefix scan
    let results = db.query("users/").limit(10).execute()?;
    for (key, value) in results {
        println!("{} = {}", 
            String::from_utf8_lossy(&key),
            String::from_utf8_lossy(&value));
    }
    ```
    </CodeGroup>
  </Step>

  <Step title="Close the database">
    Always close the database when you're done:

    <CodeGroup>
    ```python Python
    db.close()
    ```

    ```typescript Node.js
    await db.close();
    ```

    ```go Go
    db.Close()  // Or use defer db.Close() after opening
    ```

    ```rust Rust
    // Database is automatically closed when it goes out of scope
    // Or call drop(db) explicitly
    ```
    </CodeGroup>
  </Step>
</Steps>

## Complete example

Here's a complete working example:

<CodeGroup>
```python Python
from sochdb import Database

# Open database
db = Database.open("./my_db")

# Basic operations
db.put(b"greeting", b"Hello, SochDB!")
value = db.get(b"greeting")
print(value.decode())  # "Hello, SochDB!"

# Hierarchical data
db.put_path("users/alice/name", b"Alice Smith")
db.put_path("users/alice/email", b"alice@example.com")

name = db.get_path("users/alice/name")
print(name.decode())  # "Alice Smith"

# Transactions
with db.transaction() as txn:
    txn.put(b"counter", b"100")
    txn.put(b"updated", b"true")

# Clean up
db.close()
```

```typescript Node.js
import { Database } from '@sochdb/sochdb';

async function main() {
  // Open database
  const db = await Database.open('./my_db');

  // Basic operations
  await db.put('greeting', 'Hello, SochDB!');
  const value = await db.get('greeting');
  console.log(value?.toString());  // "Hello, SochDB!"

  // Hierarchical data
  await db.putPath(['users', 'alice', 'name'], 'Alice Smith');
  await db.putPath(['users', 'alice', 'email'], 'alice@example.com');

  const name = await db.getPath(['users', 'alice', 'name']);
  console.log(name?.toString());  // "Alice Smith"

  // Query prefix
  const results = await db.query('users/').limit(10).execute();
  for (const [key, value] of results) {
    console.log(`${key} = ${value.toString()}`);
  }

  // Clean up
  await db.close();
}

main().catch(console.error);
```

```go Go
package main

import (
    "fmt"
    "log"
    sochdb "github.com/sochdb/sochdb-go"
)

func main() {
    // Open database
    db, err := sochdb.Open("./my_db")
    if err != nil {
        log.Fatal(err)
    }
    defer db.Close()

    // Basic operations
    db.Put("greeting", []byte("Hello, SochDB!"))
    value, _ := db.Get("greeting")
    fmt.Println(string(value))  // "Hello, SochDB!"

    // Hierarchical data
    db.PutPath([]string{"users", "alice", "name"}, []byte("Alice Smith"))
    db.PutPath([]string{"users", "alice", "email"}, []byte("alice@example.com"))

    name, _ := db.GetPath([]string{"users", "alice", "name"})
    fmt.Println(string(name))  // "Alice Smith"

    // Query prefix
    results, _ := db.Query("users/").Limit(10).Execute()
    for _, kv := range results {
        fmt.Printf("%s = %s\n", kv.Key, string(kv.Value))
    }
}
```

```rust Rust
use sochdb::Database;
use anyhow::Result;

fn main() -> Result<()> {
    // Open database
    let db = Database::open("./my_db")?;

    // Basic operations
    db.put(b"greeting", b"Hello, SochDB!")?;
    if let Some(value) = db.get(b"greeting")? {
        println!("{}", String::from_utf8_lossy(&value));  // "Hello, SochDB!"
    }

    // Hierarchical data
    db.put_path(&["users", "alice", "name"], b"Alice Smith")?;
    db.put_path(&["users", "alice", "email"], b"alice@example.com")?;

    if let Some(name) = db.get_path(&["users", "alice", "name"])? {
        println!("{}", String::from_utf8_lossy(&name));  // "Alice Smith"
    }

    // Query prefix
    let results = db.query("users/").limit(10).execute()?;
    for (key, value) in results {
        println!("{} = {}", 
            String::from_utf8_lossy(&key),
            String::from_utf8_lossy(&value));
    }

    Ok(())
}
```
