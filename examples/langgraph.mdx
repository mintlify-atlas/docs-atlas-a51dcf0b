## Overview

Integrate SochDB with LangGraph to build AI agents that have:
- **Persistent memory** across sessions
- **Semantic retrieval** from knowledge bases
- **Context management** with token budgeting
- **State persistence** in a single database

## Use Case

A LangGraph agent that:
- Maintains conversation history
- Retrieves relevant knowledge for each query
- Manages user preferences
- Enforces token budgets for context

### Create LangGraph Agent with SochDB

```python langgraph_sochdb.py
import os
import json
import time
from typing import TypedDict, List, Dict, Annotated
from dataclasses import dataclass
import numpy as np

from langgraph.graph import StateGraph, END
from sochdb import VectorIndex, Database

# Agent State
class AgentState(TypedDict):
    """LangGraph agent state."""
    messages: List[Dict[str, str]]
    context: str
    user_id: str
    retrieved_docs: List[Dict]
    next_action: str

@dataclass
class ContextSection:
    """A section of context for the LLM."""
    name: str
    content: str
    priority: int
    tokens: int

class SochDBMemory:
    """
    SochDB-backed memory for LangGraph agents.
    
    Provides:
    - Vector storage for semantic search
    - KV storage for user preferences and state
    - Context queries with token budgeting
    """
    
    def __init__(self, db_path: str = "langgraph_memory.db", dimension: int = 768):
        self.dimension = dimension
        
        # Vector index for knowledge
        self.index = VectorIndex(
            dimension=dimension,
            max_connections=16,
            ef_construction=100
        )
        
        # KV database for state
        self.db = Database.open(db_path)
        
        self.documents = {}
        self.next_id = 0
    
    def add_document(self, content: str, embedding: np.ndarray, metadata: Dict = None):
        """Add a document with its embedding."""
        doc_id = self.next_id
        self.next_id += 1
        
        self.documents[doc_id] = {
            "id": doc_id,
            "content": content,
            "metadata": metadata or {}
        }
        
        ids = np.array([doc_id], dtype=np.uint64)
        vectors = embedding.reshape(1, -1).astype(np.float32)
        self.index.insert_batch(ids, vectors)
        
        return doc_id
    
    def search(self, query_embedding: np.ndarray, k: int = 5) -> List[Dict]:
        """Search for similar documents."""
        results = self.index.search(query_embedding.astype(np.float32), k=k)
        
        docs = []
        for doc_id, distance in results:
            if doc_id in self.documents:
                doc = self.documents[doc_id].copy()
                doc["distance"] = distance
                docs.append(doc)
        
        return docs
    
    def build_context(
        self,
        query_embedding: np.ndarray,
        user_id: str,
        token_budget: int = 4000
    ) -> str:
        """
        Build prioritized context for LLM with token budgeting.
        
        Priority 0: System prompt (always included)
        Priority 1: User preferences
        Priority 2: Retrieved documents
        """
        sections = []
        
        # Priority 0: System prompt
        system_prompt = "You are a helpful AI assistant with access to a knowledge base."
        sections.append(ContextSection(
            name="system",
            content=system_prompt,
            priority=0,
            tokens=len(system_prompt.split())
        ))
        
        # Priority 1: User preferences (from KV store)
        user_prefs = self._get_user_prefs(user_id)
        if user_prefs:
            prefs_text = f"User preferences: {json.dumps(user_prefs)}"
            sections.append(ContextSection(
                name="user_prefs",
                content=prefs_text,
                priority=1,
                tokens=len(prefs_text.split())
            ))
        
        # Priority 2: Retrieved documents
        docs = self.search(query_embedding, k=5)
        for i, doc in enumerate(docs):
            doc_text = f"[Doc {i+1}]: {doc['content']}"
            sections.append(ContextSection(
                name=f"doc_{i}",
                content=doc_text,
                priority=2,
                tokens=len(doc_text.split())
            ))
        
        # Assemble context (respecting token budget)
        sections.sort(key=lambda s: s.priority)
        
        context_parts = []
        current_tokens = 0
        
        for section in sections:
            if current_tokens + section.tokens <= token_budget:
                context_parts.append(section.content)
                current_tokens += section.tokens
        
        return "\n\n".join(context_parts)
    
    def _get_user_prefs(self, user_id: str) -> Dict:
        """Get user preferences from KV store."""
        path = f"users/{user_id}/prefs".encode()
        with self.db.transaction() as txn:
            value = txn.get(path)
            if value:
                return json.loads(value.decode())
        return {}
    
    def save_user_pref(self, user_id: str, prefs: Dict):
        """Save user preferences to KV store."""
        path = f"users/{user_id}/prefs".encode()
        with self.db.transaction() as txn:
            txn.put(path, json.dumps(prefs).encode())

def mock_embedding(text: str, dimension: int = 768) -> np.ndarray:
    """Generate a mock embedding for testing."""
    np.random.seed(hash(text) % 2**32)
    embedding = np.random.randn(dimension).astype(np.float32)
    embedding /= np.linalg.norm(embedding)
    return embedding

def create_agent_graph(memory: SochDBMemory):
    """
    Create a LangGraph agent with SochDB memory.
    """
    
    def retrieve_context(state: AgentState) -> AgentState:
        """Retrieve relevant context from SochDB."""
        user_query = state["messages"][-1]["content"]
        query_embedding = mock_embedding(user_query)
        
        # Build context with token budget
        context = memory.build_context(
            query_embedding=query_embedding,
            user_id=state["user_id"],
            token_budget=2000
        )
        
        # Get retrieved docs for state
        docs = memory.search(query_embedding, k=3)
        
        return {
            **state,
            "context": context,
            "retrieved_docs": docs,
            "next_action": "generate"
        }
    
    def generate_response(state: AgentState) -> AgentState:
        """Generate response using context."""
        # In production, this would call an LLM with state["context"]
        docs = state["retrieved_docs"]
        if docs:
            answer = f"Based on the knowledge base: {docs[0]['content']}"
        else:
            answer = "I don't have information about that."
        
        state["messages"].append({"role": "assistant", "content": answer})
        state["next_action"] = "end"
        return state
    
    def should_end(state: AgentState) -> str:
        """Determine next node."""
        return state.get("next_action", "end")
    
    # Build graph
    workflow = StateGraph(AgentState)
    
    # Add nodes
    workflow.add_node("retrieve", retrieve_context)
    workflow.add_node("generate", generate_response)
    
    # Add edges
    workflow.set_entry_point("retrieve")
    workflow.add_edge("retrieve", "generate")
    workflow.add_edge("generate", END)
    
    return workflow.compile()

def main():
    print("="*70)
    print("  SochDB + LangGraph Integration Example")
    print("="*70)
    
    # Initialize SochDB memory
    print("\n[1] Initializing SochDB memory...")
    memory = SochDBMemory(db_path="langgraph_memory.db", dimension=768)
    print("    ✓ SochDB connected")
    
    # Add knowledge documents
    print("\n[2] Indexing knowledge base...")
    knowledge = [
        {"topic": "Architecture", "content": "SochDB uses a Trie-Columnar Hybrid storage engine for O(path) lookups."},
        {"topic": "Performance", "content": "SochDB implements HNSW with SIMD acceleration for fast vector similarity search."},
        {"topic": "Features", "content": "Context queries allow building prioritized context sections with token budgets."},
        {"topic": "Integration", "content": "SochDB supports Model Context Protocol for seamless LLM tool integration."},
        {"topic": "Speed", "content": "SochDB achieves sub-millisecond search latency and 10K+ vectors/sec insert rate."}
    ]
    
    for doc in knowledge:
        embedding = mock_embedding(doc["content"])
        memory.add_document(
            content=doc["content"],
            embedding=embedding,
            metadata={"topic": doc["topic"]}
        )
    
    print(f"    ✓ Indexed {len(knowledge)} documents")
    
    # Save user preferences
    memory.save_user_pref("alice", {"language": "en", "expertise": "developer"})
    
    # Create agent graph
    print("\n[3] Building agent graph...")
    agent = create_agent_graph(memory)
    print("    ✓ Agent graph built")
    
    # Run test queries
    print("\n[4] Running test queries...")
    print("-"*70)
    
    test_queries = [
        "What storage engine does SochDB use?",
        "How fast is the vector search?",
        "What is the MCP integration?"
    ]
    
    for query in test_queries:
        # Initialize state
        state: AgentState = {
            "messages": [{"role": "user", "content": query}],
            "context": "",
            "user_id": "alice",
            "retrieved_docs": [],
            "next_action": "retrieve"
        }
        
        # Run graph
        start = time.perf_counter()
        result = agent.invoke(state)
        elapsed = (time.perf_counter() - start) * 1000
        
        print(f"\n  Q: {query}")
        print(f"  A: {result['messages'][-1]['content'][:80]}...")
        print(f"  Retrieved: {len(result['retrieved_docs'])} docs | Time: {elapsed:.1f}ms")
    
    print("\n" + "="*70)
    print("  ✅ LangGraph integration example complete!")
    print("="*70)

if __name__ == "__main__":
    main()
```

## Key Integration Points

### State Persistence

```python
# Store agent state in SochDB KV
with db.transaction() as txn:
    txn.put(f"agent/{agent_id}/state".encode(), json.dumps(state).encode())

# Restore state
with db.transaction() as txn:
    data = txn.get(f"agent/{agent_id}/state".encode())
    state = json.loads(data.decode())
```

### Context Assembly

```python
def retrieve_context(state: AgentState) -> AgentState:

    query_embedding = embed(state["messages"][-1]["content"])
```
